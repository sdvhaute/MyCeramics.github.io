        pool.query(`UPDATE projects SET thrown = true WHERE id = $1`, [projectid], (err) => {
            console.log('thrown set to true');

            console.log('thrown set to false');

            res.redirect(`/api/v1/projects/${projectid}`);
        });
    }

};
